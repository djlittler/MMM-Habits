Usage

/*
var circle = new Segmentor(
  [element or Id of svg],
  {
    totalSegments: [total number of segments],
    filledSegments: [number of filled segments],
    offset: [how much space between each segment]
  }
);
*/
// Ex:
var circle = new Segmentor('circle', {
  totalSegments: 3,
  filledSegments: 10,
  offset: 1
});
